## 3.2
```c
#include <stdio.h>

int main()
{
    int a;
    printf("Enter a number:");
    scanf("%d", &a);
    if (a < 0)
    {
        a = -a;
    }
    int b2,b1,b0;
    b2 = a / 100;
    b1 = (a - b2 * 100) / 10;
    b0 = a - b2 * 100 - b1 * 10;
    int c;
    c = b0 * 100 + b1 * 10 + b2;
    printf("%d\n", c);
    return 0;
}
```
## 3.3
```c
#include <stdio.h>

int main()
{
    int n;
    printf("请输入存款期:");
    scanf("%d", &n);
    double capital;
    printf("请输入本金:");
    scanf("%lf", &capital);
    double rate = 0.0225;
    double interest = 0;
    for(int i = 1; i <= n; i++)
    {
        interest = capital * rate;
        capital = capital + interest;
    }
    printf("本利之和为 %lf\n", capital);
    return 0;
}
```

```c
#include <stdio.h>
#include <math.h>

int main()
{
    double a,b,c,d;
    double x1,x2;
    printf("有一元二次方程，请输入x的平方的系数a:");
    scanf("%lf", &a);
    printf("请输入x的系数b:");
    scanf("%lf", &b);
    printf("请输入常数项c:");
    scanf("%lf", &c);
    d = b * b - 4 * a * c;
    d = sqrt(d);
    x1 = (-b - d) / (2 * a);
    x2 = (-b + d) / (2 * a);
    printf("方程的两个根分别为 %lf 和 %lf\n", x1, x2);
    return 0;
}
```

## 3.5
```c
#include <stdio.h>
#define PI 3.14159

int main()
{
    double r,V,S;
    printf("请输入球的半径r:");
    scanf("%lf", &r);
    V = (4.0 / 3) * PI * r * r * r;
    S = 4 * PI * r * r;
    printf("球的体积为   %lf\n", V);
    printf("球的表面积为 %lf\n", S);
    return 0;
}
```
