#include <stdio.h>
#include <math.h>

int main()
{
    printf("请输入三角形的三边：\n");
    printf("三角形三边的输入格式:3, 4, 5\n");
    double a, b, c, p, S;
    scanf("%lf, %lf, %lf", &a, &b, &c);
    if(a + b > c && a + c > b && b +c > a)
    {
        p = (a + b + c) / 2.0;
        S = sqrt(p * (p-a) * (p-b) * (p-c));
        printf("%lf\n", S);
    }
    else
    {
        printf("不能构成三角形\n");
    }
    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int main()
{
    printf("请输入本金：\n");
    double money;
    scanf("%lf", &money);
    printf("请输入期限:\n");
    int year;
    scanf("%d", &year);
    switch (year)
    {
        case 1:
            money = money * 1.0225;
            break;
        case 2:
            for (int i = 0; i < 2; i++)
            {
                money = money * 1.0243;
            }
            break;
        case 3:
            for (int i = 0; i < 3; i++)
            {
                money = money * 1.0270;
            }
            break;

            for (int i = 0; i < 5; i++)

                money = money * 1.0288;
            }
            break;

#include <stdio.h>

int main()
{
    int year, flag;
    printf("Input a year:");
    scanf("%d", &year);
    flag = (((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0)) ? 1 : 0;
    if (flag == 1)
    {
        printf("%d is a leap year.\n", year);
    }
    else
    {
        printf("%d is not a leap year.\n", year);
    }
    return 0;
}
