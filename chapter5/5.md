int main()
{
    double num;
    scanf("%lf", &num);
    if(num < 0)
    {
        num = -num;
    }
    printf("%lf\n", num);
    return 0;
}

## 5.3
```c
#include <stdio.h>
#include <math.h>

int main()
{
    printf("请输入三角形的三边：\n");
    printf("三角形三边的输入格式:3, 4, 5\n");
    double a, b, c, p, S;
    scanf("%lf, %lf, %lf", &a, &b, &c);
    if(a + b > c && a + c > b && b +c > a)
    {
        p = (a + b + c) / 2.0;
        S = sqrt(p * (p-a) * (p-b) * (p-c));
        printf("%lf\n", S);
    }
    else
    {
        printf("不能构成三角形\n");
    }
    return 0;
}
```
